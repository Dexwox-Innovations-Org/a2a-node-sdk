```typescript
  createDataMessage,
  validateMessageParts
} from '@dexwox/a2a-core';

const filePart = createFileMessage(fileData, 'image/png', 'image.png');
```


## Configuration

Configure clients and servers with comprehensive options:

```typescript
import { MessageClient, A2AServer } from '@dexwox/a2a-node';

// Client configuration
const client = new MessageClient({
  baseUrl: 'https://api.example.com',
  timeout: 10000,
  retries: 3,
  headers: {
    'Authorization': 'Bearer token'
  }
});

// Server configuration
const server = new A2AServer({
  agentCard: { /* ... */ },
  executor: myExecutor,
  middleware: [authMiddleware, loggingMiddleware],
  cors: { origin: '*' }
});
```
